% FRONT-/MAIN-/BACKMATTER ------------------------------------------------------

\newcounter{frontcounter}
\setcounter{frontcounter}{1}

\def\frontmatter{%
    \pagenumbering{roman}
    \setcounter{page}{1}
}%

\def\mainmatter{%
    \setcounter{frontcounter}{\value{page}}
    \pagenumbering{arabic}
    \setcounter{page}{1}
}%

\def\backmatter{%
    \pagenumbering{roman}
    \setcounter{page}{\value{frontcounter}}
}%
